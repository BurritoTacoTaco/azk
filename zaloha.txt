  <script>
    let scores = [0, 0];

    function increment(player) {
      scores[player - 1]++;
      updateScores();
    }

    function reset(player) {
      scores[player - 1] = 0;
      updateScores();
    }

    function resetAll() {
      scores = [0, 0];
      updateScores();
    }

    function updateScores() {
      document.getElementById('score1').textContent = scores[0];
      document.getElementById('score2').textContent = scores[1];
    }
